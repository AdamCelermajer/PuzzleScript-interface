main():
    args = parse_cli()
    cfg = Config(game=args.game_id, ...)
    dashboard = ScreenDashboard(game_id=args.game_id, mode="RUN", ...)

    env = ArcadeEnv(
        game_id=args.game_id,
        backend_url=args.backend_url,
        renderer=dashboard.render,
    )
    llm_adapter = LlmClient(cfg, event_sink=dashboard.push_event)
    engine = EngineArchitecture.from_config(cfg, llm_adapter, event_sink=...)

    loop = RuleReasoningLoop(
        env=env,
        perception=engine.perceiver,
        memory=engine.memory,
        rulebook=engine.rulebook,
        planner=engine.planner,
        inducer=engine.inducer,
        dashboard=dashboard,
    )

    dashboard.run_engine(lambda:
        loop.run(max_steps=cfg.max_steps, game_id=cfg.game)
    )

Runtime Loop

client/runtime/runner.py 
RuleReasoningLoop.run(max_steps, game_id):
    frame = env.reset()
    state = perception.perceive(frame)
    memory.append_state(state)

    while steps < max_steps:
        decision = planner.choose_action()
        predictions = rulebook.predict(state, decision.action)

        after_frame = env.step(decision.action)
        after_state = perception.perceive(after_frame)
        rulebook.record_prediction_result(state, decision.action, after_state, predictions)

        memory.append_action_result(decision.action, after_state)

        unexplained = not predictions or after_state not in predictions
        if unexplained:
            planner.clear_llm_plan()

        if decision.exploratory or unexplained:
            inducer.propose_from_memory(game_id, memory)

        handle wins, resets, game over, and unchanged boards

Perception.perceive(frame_data):
    grid = tuple(tuple(int(cell) for cell in row)
                 for row in last_grid(frame_data.frame))

    return EngineState(
        grid=grid,
        state=frame_data.state,
        levels_completed=frame_data.levels_completed,
        win_levels=frame_data.win_levels,
        game_id=frame_data.game_id,
        image=image_from(frame_data.rendered_frame),
        available_actions=tuple(frame_data.available_actions),
    )

Planner First

client/engine/planner.py 
Planner.choose_action(memory):
    current = memory.current_state()

    if active_plan exists:
        return next_action_from(active_plan)

    if no active_goal:
        active_goal = goal_manager.ensure_goal(memory)

    plan = search_rulebook(current, active_goal)
    if plan:
        active_plan = plan
        return next_action_from(active_plan)

    subgoal, action = ask_for_exploratory_step(
        current_state=current,
        recent_memory=memory.recent_context(),
        images=memory.available_images(),
    )
    active_goal = subgoal
    return PlanDecision(action, exploratory=True)

Memory And Rulebook

client/engine/memory.py + client/engine/rulebook.py 
EngineMemory.append_state(state):
    if timeline is empty:
        append StateNode(state)
    else:
        update current observed state

EngineMemory.append_action_result(action, after_state):
    append ActionEdge(previous_state, action, after_state)
    append StateNode(after_state)

EngineMemory.recent_transitions():
    derive (before_state, action, after_state)
    from adjacent timeline nodes

Rulebook.predict(state, action):
    for accepted rule matching action:
        if rule conditions match state:
            yield predicted next EngineState

Rule Induction And Verification

client/engine/induction.py + client/engine/verifier.py 
RuleInducer.propose_from_memory(game_id, memory):
    events, images = memory.latest_visual_context()
    evidence = memory.recent_transitions(1)
    known_rules = rulebook.known_rules_text()
    # known_rules is natural-language summaries only

    data = llm.call_json(
        induction_prompt(game_id, evidence, known_rules),
        image_data_urls=images,
    )
    candidates = parse GeneralizedRule objects

    for candidate in candidates:
        result = verifier.verify(candidate)
        if result.accepted:
            rulebook.add_generalized_rule(result.rule)
        else:
            log("Rejected rule candidate", result.failures)

RuleVerifier.verify(candidate):
    for evidence in candidate.evidence_ids:
        transition = memory.transition_by_id(evidence)
        predicted = candidate.predict(transition.before)
        if transition.after not in predicted:
            return VerificationResult(candidate, failures=[...])

    return VerificationResult(candidate, failures=[])
