legendes:
{
  "P": "player",
  "#": "wall",
  ".": "empty space",
  "*": "crate",
  "O": "target"
}

final_goal:
Move all crates to the targets

"Movement":{
IF action is 'DIR' AND '.' is in DIR of P THEN 'P' moves DIR,
IF action is 'DIR' AND 'O' is in DIR of P THEN 'P' moves DIR,
IF action is 'DIR' AND '*' is in DIR of P AND cell in DIR of '*' is '.' THEN 'P' and '*' both move DIR
}

"Collision":{
IF action is 'W' AND '#' is above P THEN 'P' does not move,
IF action is 'DIR' AND '*' is in DIR of P AND cell in DIR of '*' is '#' THEN 'P' does not move
}
