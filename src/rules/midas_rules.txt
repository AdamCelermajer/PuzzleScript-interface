legendes:
{
  "P": "Player",
  "C": "Player jumping",
  "#": "Wall",
  "G": "Gold",
  "D": "Dirt",
  "T": "Target",
  "W": "Platform",
  "L": "Exit Door",
  ".": "Empty space"
}

final_goal:
Reach the target (T) by moving and jumping, turning touched solid blocks into dirt and then gold, to spawn the exit door (L) and advance to the next level.

"Movement":{
IF action is '<WASD>' AND the cell in the direction of movement is '.' THEN P moves in the direction of movement,
IF action is 'D' AND the cell to the right of C is '.' THEN C moves right
}

"State Change":{
IF action is 'D' AND the cell to the right of P is '.' AND the cell below the cell to the right of P is '#' THEN '#' becomes 'D',
IF action is '<WASD>' AND the cell below P is 'D' THEN 'D' becomes 'G',
IF action is '<WASD>' AND the cell to the right of P is 'D' THEN 'D' becomes 'G',
IF action is 'D' AND the cell to the left of P is 'D' THEN 'D' becomes 'G',
IF action is 'W' AND the cell above P is '.' THEN P becomes C,
IF action is 'D' AND the cell to the right of C is '.' THEN C becomes P,
IF action is 'D' AND the cell to the right of C is '.' AND the cell to the right of the cell to the right of C is '#' THEN '#' becomes 'D',
IF action is 'D' AND the cell to the right of C is '.' AND the cell below the cell to the right of C is '#' THEN '#' becomes 'D',
IF action is 'D' AND the cell to the right of P is '.' AND the cell below the cell to the right of P is '.' AND the cell above the cell to the right of P is '.' THEN '.' becomes 'L',
IF action is 'D' AND the cell to the right of P is '.' AND the cell below the cell to the right of P is 'W' AND the cell above the cell to the right of P is '.' THEN '.' becomes 'L',
IF action is 'W' AND the cell above P is '.' AND the cell to the right of the cell above P is '#' THEN '#' becomes 'D'
}

"Collision":{
IF action is 'D' AND the cell to the right of P is 'D' THEN P does not move,
IF action is 'D' AND the cell to the right of P is 'G' THEN P does not move,
IF action is 'D' AND the cell to the right of C is 'G' THEN C does not move
}
